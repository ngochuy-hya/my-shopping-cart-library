Sample Code

import React from 'react';
import { ShoppingCart, useShoppingCart } from 'my-shopping-cart-library';
import 'my-shopping-cart-library/dist/index.css';

function App() {
  const {
    cart,
    addItem,
    updateItem,
    removeItem,
    clearCart
  } = useShoppingCart();

  const sampleProduct = {
    id: '1',
    name: 'Sample Product',
    price: 29.99,
    image: 'https://via.placeholder.com/60x60',
    description: 'A great sample product'
  };

  return (
    <div className="App">
      <button onClick={() => addItem(sampleProduct)}>
        Add Sample Product
      </button>
      
      <ShoppingCart 
        items={cart.items}
        onUpdateItem={updateItem}
        onRemoveItem={removeItem}
        onClearCart={clearCart}
        currency="$"
      />
    </div>
  );
}

export default App;
